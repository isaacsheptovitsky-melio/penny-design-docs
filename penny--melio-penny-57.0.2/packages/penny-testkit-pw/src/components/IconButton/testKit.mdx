# IconButton Playwright Test Kit
---

Testing utilities for the `IconButton` component that provide easy access to component interactions, states, and accessibility features for Playwright testing.

## Import

```typescript
import { createIconButtonTestKit } from '@melio/penny-testkit-pw';
```


## Methods

<Markdown>
{`
| Method | Description | Return value |
|--------|-------------|--------------|
| \`dataTestId\` | The data-testid used to identify the component | \`string\` |
| \`getElement()\` | Direct access to the DOM element | \`Locator\` |
| \`click()\` | Clicks on the element | \`Promise<void>\` |
| \`getDisabled()\` | Checks if the element is disabled | \`Promise<boolean>\` |
| \`getIsLink()\` | Checks if the element is a link | \`Promise<boolean>\` |
| \`getHref()\` | Gets the element's href | \`Promise<string | null>\` |
`}
</Markdown>

## Examples

```typescript
import { test, expect } from '@playwright/test';
import { createIconButtonTestKit } from '@melio/penny-testkit-pw';

test.describe('IconButton Component', () => {
  test('should be link', async ({ page }) => {
    const testKit = createIconButtonTestKit(page, { dataTestId: 'custom-datatestid' });
    
    await page.goto('/some-page');

    expect(await testKit.getIsLink()).toBe(true);
  });

  test("shouldn't be disabled", async ({ page }) => {
    const testKit = createIconButtonTestKit(page, { dataTestId: 'custom-datatestid' });
    
    await page.goto('/some-page');
    
    expect(await testKit.getDisabled()).toBe(false);
  });

  test('should get aria-label attribute', async ({ page }) => {
    const testKit = createIconButtonTestKit(page, { dataTestId: 'custom-datatestid' });
    
    await page.goto('/some-page');
    
    expect(await testKit.getElement().getAttribute()'aria-label').toBe('Upload file');
  });
});
```
