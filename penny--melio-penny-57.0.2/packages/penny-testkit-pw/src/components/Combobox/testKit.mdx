# Combobox Playwright Test Kit
---

Testing utilities for the `Combobox` component that provide easy access to component interactions, states, and accessibility features for Playwright testing.

## Import

```typescript
import { createComboboxTestKit } from '@melio/penny-testkit-pw';
```

## Methods

<Markdown>
  {`
| Method | Description | Return value |
|--------|-------------|--------------|
| \`dataTestId\` | The data-testid used to identify the component | \`string\` |
| \`getElement()\` | Direct access to the root element | \`Locator\` |
| \`getIsDisabled()\` | Checks if the component is disabled | \`Promise<boolean>\` |
| \`getIsReadOnly()\` | Checks if the component is read only | \`Promise<boolean>\` |
| \`getIsRequired()\` | Checks if the component is required | \`Promise<boolean>\` |
| \`getIsInvalid()\` | Checks if the component is invalid | \`Promise<boolean>\` |
| \`getIsViewMode()\` | Checks if the component is view mode | \`Promise<boolean>\` |
| \`getIsMenuOpen()\` | Checks if the component is menu open | \`Promise<boolean>\` |
| \`getValue()\` | Gets the selected value | \`Promise<string>\` |
| \`getInputValue()\` | Gets the input value | \`Promise<string>\` |
| \`getPlaceholder()\` | Gets the placeholder | \`Promise<string | null>\` |
| \`clickTrigger()\` | Clicks the element | \`Promise<void>\` |
| \`type(text: string)\` | Types text into the element | \`Promise<void>\` |
| \`clear()\` | Clears the element value | \`Promise<void>\` |
| \`clickOutside()\` | Clicks the outside | \`Promise<void>\` |
| \`clickClearButton()\` | Clicks the clear button | \`Promise<void>\` |
| \`clickOptionByText(text: string)\` | Clicks the option by text | \`Promise<void>\` |
| \`clickOptionByIndex(index: number)\` | Clicks the option by index | \`Promise<void>\` |
| \`clickOptionByTestId(optionTestId: string)\` | Clicks the option by test id | \`Promise<void>\` |
`}
</Markdown>

## Examples

```typescript
import { test, expect } from '@playwright/test';
import { createComboboxTestKit } from '@melio/penny-testkit-pw';


test('should be disabled', async ({ page }) => {
  const testKit = createComboboxTestKit(page, { dataTestId: 'custom-data-testid' });
  await page.goto('/combobox-demo');

  expect(await testKit.getIsDisabled()).toBe(true);
});

test('should have value when selecting an option', async ({ page }) => {
  const testKit = createComboboxTestKit(page, { dataTestId: 'custom-data-testid' });
  await page.goto('/combobox-demo');

  await testKit.clickTrigger();
  await testKit.type('Los');
  expect(await testKit.getInputValue()).toBe('Los');

  await testKit.clickOptionByText('Los Angeles');
  expect(await testKit.getValue()).toBe('Los Angeles');
});
```
