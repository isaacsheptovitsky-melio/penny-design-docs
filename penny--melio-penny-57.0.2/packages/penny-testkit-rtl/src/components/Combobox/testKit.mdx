# Combobox RTL Test Kit
---

Testing utilities for the `Combobox` component that provide easy access to component interactions, states, and accessibility features for RTL (React Testing Library).

## Import

```typescript
import { createComboboxTestKit } from '@melio/penny-testkit-rtl';
```

## Methods

<Markdown>
  {`
| Method | Description | Return value |
|--------|-------------|--------------|
| \`dataTestId\` | The data-testid used to identify the component | \`string\` |
| \`getElement()\` | Direct access to the root element | \`HTMLElement\` |
| \`getIsDisabled()\` | Checks if the component is disabled | \`boolean\` |
| \`getIsReadOnly()\` | Checks if the component is read only | \`boolean\` |
| \`getIsRequired()\` | Checks if the component is required | \`boolean\` |
| \`getIsInvalid()\` | Checks if the component is invalid | \`boolean\` |
| \`getIsViewMode()\` | Checks if the component is view mode | \`boolean\` |
| \`getIsMenuOpen()\` | Checks if the component is menu open | \`boolean\` |
| \`getValue()\` | Gets the selected value | \`string\` |
| \`getInputValue()\` | Gets the input value | \`string\` |
| \`getPlaceholder()\` | Gets the placeholder | \`string | null\` |
| \`clickTrigger(options?: Options)\` | Clicks the element | \`Promise<void>\` |
| \`type(text: string, options?: Options)\` | Types text into the element | \`Promise<void>\` |
| \`clear()\` | Clears the element value | \`Promise<void>\` |
| \`clickOutside(options?: Options)\` | Clicks the outside | \`Promise<void>\` |
| \`clickClearButton(options?: Options)\` | Clicks the clear button | \`Promise<void>\` |
| \`clickOptionByText(text: string, options?: Options)\` | Clicks the option by text | \`Promise<void>\` |
| \`clickOptionByIndex(index: number, options?: Options)\` | Clicks the option by index | \`Promise<void>\` |
| \`clickOptionByTestId(optionTestId: string, options?: Options)\` | Clicks the option by test id | \`Promise<void>\` |
`}
</Markdown>

## Examples

```typescript
import { render } from '@testing-library/react';
import { Combobox } from '@melio/penny';
import { createComboboxTestKit } from '@melio/penny-testkit-rtl';

describe('Combobox', () => {
  it('should be disabled', () => {
    const testKit = createComboboxTestKit({ dataTestId: 'custom-data-testid' });
    render(<Combobox data-testid="custom-data-testid" isDisabled />);

    expect(testKit.getIsDisabled()).toBe(true);
  });

  it('should have value when selecting an option', async () => {
    const testKit = createComboboxTestKit({ dataTestId: 'custom-data-testid' });
    render(<Combobox data-testid="custom-data-testid" />);

    await testKit.clickTrigger();
    await testKit.type('Los');
    expect(testKit.getInputValue()).toBe('Los');

    await testKit.clickOptionByText('Los Angeles');
    expect(testKit.getValue()).toBe('Los Angeles');
  });
});
```
