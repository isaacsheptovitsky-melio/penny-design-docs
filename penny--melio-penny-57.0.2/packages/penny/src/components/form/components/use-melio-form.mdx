# Form - `useMelioForm`

The `useMelioForm` hook is a one-stop-shop hook to create all the logic of the form.<br />

### Generic
The hook receives a generic that tells the structure of the form data.

```tsx
import { useMelioForm } from '@melio/penny';

type MyFormFields = {
  vendorId: string;
  billAmount: string;
  billNumber: string;
  dueDate: Date;
  note: string;
};

const { /* ... */ } = useMelioForm<MyFormFields>({ /* ... */ });
```

### Usage

```tsx
// MyForm.tsx
import { Button, Form, FormField, useMelioForm } from '@melio/penny';

export type MyFormFields = {
  vendorId: string;
  billAmount: string;
  billNumber: string;
  dueDate: Date;
  note: string;
};

type MyFormProps = {
  onSubmit: (data: MyFormFields) => void;
}

export const MyForm: VFC<MyFormProps> = ({ onSubmit }) => {
  const {
    formProps,
    registerField,
    submitButtonProps,
    cancelButtonProps
  } = useMelioForm<MyFormFields>({ onSubmit });

  return (
    <Group variant="vertical">
       <Form columns={2} {...formProps}>
          <FormField
            {...registerField('vendorId')}
            colSpan={2}
            isRequired
            labelProps={{ label: "Vendor's business name" }}
            render={(props) => <SelectNew {...props} placeholder="Search or add a vendor" options={vendorOption} />}
          />

          <FormField
            {...registerField('billAmount')}
            colSpan={2}
            isRequired
            labelProps={{ label: 'Bill amount' }}
            render={(props) => <AmountField {...props} placeholder="$0.00" />}
          />

          <FormField
            {...registerField('billNumber')}
            labelProps={{ label: 'Invoice #' }}
            render={(props) => <TextField {...props} placeholder="e.g. 200000" />}
          />

          <FormField
            {...registerField('dueDate')}
            isRequired
            labelProps={{ label: 'Due date' }}
            render={(props) => <DateField {...props} placeholder="Select due date" />}
          />

          <FormField
            {...registerField('note')}
            colSpan={2}
            labelProps={{ label: 'Note to self' }}
            render={(props) => <TextField {...props} placeholder="e.g. invoice #, office expenses" />}
          />
       </Form>

       <Button {...submitButtonProps} label="Submit"></Button>
       <Button {...cancelButtonProps} label="Cancel"></Button>
    </Group>
  );
}
```
